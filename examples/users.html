// Mockup code
function ajax(url, callback){
  var fakeusers = [
    { name: 'Tom',    points: 15, picture: 'apple.jpg' },
    { name: 'Laura',  points: 43, picture: 'banana.jpg' },
    { name: 'Peter',  points: 28, picture: 'egg.jpg' },
    { name: 'Martha', points: 14, picture: 'strawberry.jpg' },
    { name: 'Sarah',  points: 76, picture: 'orange.jpg' },
    { name: 'John',   points: 16, picture: 'pear.jpg' }
    ];
  callback.call(this, fakeusers);
}

// Load the user list through ajax
ajax('api/path/users', function(users){

  // Load the users templates
  template('template.user', users, function(user){

    // Fill the template with data
    this.querySelector('.name').innerHTML = user.name;
    this.querySelector('.points').innerHTML = user.points;
    this.querySelector('img').setAttribute('src', 'img/' + user.picture);
  }).into('.users');
});
